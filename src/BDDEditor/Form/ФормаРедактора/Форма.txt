			КонецЕсли;	
		Иначе
			Оповещение = Новый ОписаниеОповещения("ПослеЗакрытияВопроса",ЭтотОбъект,ПроцедураПослеСохранения);

		КонецЕсли;
	Иначе	
		Если ПроцедураПослеСохранения<>"" Тогда
